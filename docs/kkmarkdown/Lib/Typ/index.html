Type definition

type stack_elt =
  1. | Em
  2. | Strong
  3. | EmStrong
  4. | Strike
  5. | Code
type span =
  1. | A of string
  2. | Br
  3. | CharSpan of char
  4. | CharsSpan of string
  5. | StackOpen of stack_elt
  6. | StackClose of stack_elt
  7. | UnicodeSpan of string
  8. | UnsafeA of {
    1. spans : span list;
    2. link : string;
    }
type list_typ =
  1. | Ordered
  2. | Unordered
type block =
  1. | CodeBlock of string list
  2. | Details of {
    1. title : span list;
    2. body : t;
    }
  3. | H1 of span list
  4. | H2 of span list
  5. | H3 of span list
  6. | H4 of span list
  7. | H5 of span list
  8. | H6 of span list
  9. | Hr
  10. | List_ of list_typ * li list
  11. | P of span list
  12. | Quote of t
  13. | UnsafeCodeBlock of {
    1. cb : string list;
    2. classes : string list;
    }
  14. | UnsafeImg of {
    1. alt : string;
    2. link : string;
    3. classes : string list;
    }
  15. | UnsafeInlineHtml of string list
and li =
  1. | Li of span list
  2. | LiP of block list
and t = block list
